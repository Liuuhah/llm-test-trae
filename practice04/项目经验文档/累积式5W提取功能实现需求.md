# 累积式 5W 提取功能实现需求文档

## 📋 任务概述

为 `chat_compress_client.py` 实现**累积式 5W 信息提取功能**，解决当前设计中"提取失败导致对话丢失"的问题。

---

## 🎯 核心需求

### **问题背景**

当前设计：
- 每 5 轮对话触发一次 5W 提取
- 每次只提取最近 10 条消息（`chat_history[-10:]`）
- 如果 LLM 认为"信息太少"而返回"无法提取"，这 10 条消息就永远丢失了

**举例：**
```
第 1-5 轮 → 提取失败 → ❌ 永久丢失
第 6-10 轮 → 提取成功 → ✅ 保存
```

### **期望行为**

实现**累积式提取**：
- 如果某次提取失败，记录起始位置
- 下次提取时，从上次失败的位置开始，包含所有未成功提取的对话
- 但为了节省 token，采用**分级处理策略**：
  - **历史对话**（之前失败的）：生成 1-2 句话的简要摘要
  - **当前对话**（最近的 5 轮）：完整提取 5W 信息

**举例：**
```
第 1-5 轮 → 提取失败 → 记录 start_index=0
第 1-10 轮 → 混合提取：
  - 第 1-5 轮 → 简要摘要："用户与AI互相问候，用户自我介绍姓刘"
  - 第 6-10 轮 → 完整 5W：Who: 刘先生, What: 查询成都天气...
  → ✅ 保存成功，重置 start_index
```

```python
def _extract_5w_info(self):
    """执行 5W 关键信息提取（支持累积模式）"""
    
    # 判断是否为累积模式
    if self.last_failed_start_index is not None:
        # 累积模式：从上次失败的位置开始提取
        recent_messages = self.chat_history[self.last_failed_start_index:]
        is_cumulative = True
        print(f"[关键信息提取] 🔄 累积模式：提取 {len(recent_messages)} 条消息（包含上次失败的内容）")
    else:
        # 正常模式：提取最近 extract_interval * 2 条消息
        recent_messages = self.chat_history[-(self.extract_interval * 2):]
        is_cumulative = False
        print(f"[关键信息提取] 标准模式：提取最近 {len(recent_messages)} 条消息")
    
    # 构建提取提示词（传入累积模式标志）
    extract_prompt = self._build_extract_prompt(recent_messages, is_cumulative)
    
    # 调用 LLM 进行提取
    print("[关键信息提取] 正在调用 LLM...")
    response_data = self._send_extract_request(
        extract_prompt, 
        max_tokens=4096,  # 增加 token 限制，支持更长的输出
        temperature=0.5
    )
    
    # 解析响应
    extracted_info, status, reason = self._parse_extraction_response(response_data)
    
    # 根据结果更新状态
    if status == "success":
        # 提取成功，重置累积状态
        if is_cumulative:
            print(f"[关键信息提取] ✅ 累积提取成功，已覆盖之前的失败记录")
        self.last_failed_start_index = None
        
        # 保存到日志文件
        self._save_extraction_to_log(extracted_info)
        
    elif status == "insufficient_info":
        # 提取失败（信息不足），记录起始位置
        if not is_cumulative:
            # 第一次失败，记录起始位置
            self.last_failed_start_index = len(self.chat_history) - len(recent_messages)
            print(f"[关键信息提取] ⚠️ 首次失败，记录起始位置: index={self.last_failed_start_index}")
        else:
            # 累积模式仍然失败，保持起始位置不变
            print(f"[关键信息提取] ⚠️ 累积提取仍失败，继续累积")
        
        # 保存失败记录
        self._save_failed_extraction(
            round_number=len(self.chat_history) // 2,
            reason=reason,
            recent_messages=recent_messages
        )
        
    elif status == "technical_error":
        # 技术故障，不改变累积状态
        print(f"[关键信息提取] ❌ 技术故障，保持当前状态")
```

### **4. 修改 `_parse_extraction_response()` 方法**

**位置：** 约第 1212 行

**需要支持两种格式的解析：**

```python
def _parse_extraction_response(self, response_data):
    """解析 LLM 返回的提取结果
    
    Returns:
        tuple: (extracted_info, status, reason)
    """
    
    # 1. 检查响应是否为空（技术故障）
    if response_data is None:
        return None, "technical_error", "API 调用失败，请检查网络连接"
    
    if 'choices' not in response_data or len(response_data['choices']) == 0:
        return None, "technical_error", "LLM 返回空响应，可能是 token 不足或模型错误"
    
    # 2. 提取内容
    choice = response_data['choices'][0]
    message = choice.get('message', {})
    
    # 优先使用 content 字段（最终结果），而不是 reasoning_content（思考过程）
    content = message.get('content', '').strip()
    
    # 如果 content 为空，才尝试从 reasoning_content 提取
    if not content:
        reasoning_content = message.get('reasoning_content', '')
        
        if reasoning_content:
            # 从 reasoning_content 中提取最后的 5W 结果
            who_match = re.search(r'-\s*Who:', reasoning_content)
            if who_match:
                content = reasoning_content[who_match.start():].strip()
            else:
                lines = reasoning_content.strip().split('\n')
                content = '\n'.join(lines[-5:]).strip()
    
    # 如果仍然为空，技术故障
    if not content:
        delta = choice.get('delta', {})
        content = delta.get('content', '').strip()
    
    if not content:
        return None, "technical_error", "LLM 返回空内容，无法提取信息"
    
    # 3. 检查是否为累积模式的混合格式
    if '=== 历史对话摘要 ===' in content and '=== 当前对话 5W 信息 ===' in content:
        # 累积模式：提取 5W 部分
        five_w_section = content.split('=== 当前对话 5W 信息 ===')[1].strip()
        
        # 检查 5W 部分是否可提取
        if '【无法提取】' in five_w_section:
            # 即使历史有摘要，但当前对话仍无法提取
            return None, "insufficient_info", "当前对话内容可提取的信息太少"
        
        # 验证 5W 格式
        required_keywords = ['Who', 'What']
        has_required = any(keyword in five_w_section for keyword in required_keywords)
        
        if not has_required:
            return None, "format_error", f"5W 部分不符合格式要求。内容: {five_w_section[:200]}"
        
        # 提取成功，返回完整内容（包含历史摘要和 5W）
        print(f"[解析成功] 累积模式：提取到 {len(content)} 字符（含历史摘要）")
        return content, "success", None
    
    # 4. 检查 LLM 是否标记为"无法提取"（标准模式）
    if "【无法提取】" in content:
        reason_match = re.search(r'【无法提取】(.+)', content)
        reason = reason_match.group(1).strip() if reason_match else "对话内容无法提取"
        return None, "insufficient_info", reason
    
    # 5. 验证是否包含 5W 关键字段（标准模式）
    required_keywords = ['Who', 'What']
    has_required = any(keyword in content for keyword in required_keywords)
    
    if not has_required:
        if len(content) < 50:
            return None, "insufficient_info", "返回内容过短，无法提取有效信息"
        else:
            print(f"[调试] 返回内容前200字符: {content[:200]}")
            return None, "format_error", f"返回内容不符合 5W 格式要求（缺少 Who/What 字段）。返回内容长度: {len(content)} 字符"
    
    # 6. 检查"未提及"数量
    unmentioned_count = content.count("未提及")
    if unmentioned_count >= 4:
        return None, "insufficient_info", "对话内容可提取的信息太少，没有值得提取的关键信息"
    
    # 7. 提取成功
    print(f"[解析成功] 标准模式：提取到 {len(content)} 字符的 5W 结果")
    return content, "success", None
```

```python
def _save_extraction_to_log(self, extracted_info):
    """将提取的关键信息保存到日志文件"""
    try:
        log_dir = r"D:\chat-log"
        os.makedirs(log_dir, exist_ok=True)
        log_file = os.path.join(log_dir, "log.txt")
        
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        
        # 检测是否为累积模式
        is_cumulative = '=== 历史对话摘要 ===' in extracted_info
        
        with open(log_file, 'a', encoding='utf-8') as f:
            f.write(f"\n{'='*60}\n")
            f.write(f"📅 提取时间: {timestamp}\n")
            if is_cumulative:
                f.write(f"🔄 提取模式: 累积模式（包含历史对话）\n")
            else:
                f.write(f"📊 提取模式: 标准模式\n")
            f.write(f"{'='*60}\n\n")
            f.write(extracted_info)
            f.write(f"\n{'='*60}\n")
        
        print(f"[关键信息提取] 💾 已保存到: {log_file}")
        
    except Exception as e:
        print(f"[关键信息提取] ❌ 保存失败: {str(e)}")
```
